# -*- coding: utf-8 -*-
# 基础求和算法模块

import numpy as np

def sum_basic(iterable):
    """
    普通求和算法
    
    参数:
        iterable: 可迭代对象，包含数字
    
    返回:
        求和结果
    """
    result = 0
    for num in iterable:
        result += num
    return result

def sum_numpy(array):
    """
    使用NumPy进行向量化求和
    
    参数:
        array: NumPy数组或可转换为数组的序列
    
    返回:
        求和结果
    """
    return np.sum(np.asarray(array))

def sum_with_stats(iterable):
    """
    求和并返回统计信息
    
    参数:
        iterable: 可迭代对象，包含数字
    
    返回:
        字典，包含sum, count, mean, variance, std
    """
    total = 0
    count = 0
    sum_sq = 0
    
    for num in iterable:
        total += num
        sum_sq += num ** 2
        count += 1
    
    if count == 0:
        return {'sum': 0, 'count': 0, 
                'mean': 0, 'variance': 0, 'std': 0}
    
    mean = total / count
    variance = sum_sq / count - mean ** 2
    std = variance ** 0.5
    
    return {
        'sum': total,
        'count': count,
        'mean': mean,
        'variance': variance,
        'std': std
    }

def sum_range(start, end, step=1):
    """
    计算等差数列的和
    
    参数:
        start: 起始值
        end: 结束值(不包含)
        step: 步长，默认为1
    
    返回:
        等差数列求和结果
    """
    if start >= end:
        return 0
    
    count = ((end - start - 1) // step) + 1
    return count * (start + (start + (count - 1) * step)) // 2

def sum_kahan(iterable):
    """
    Kahan求和算法 - 减少浮点精度损失
    
    算法原理:
        通过维护一个补偿变量来存储舍入误差，
        在累加过程中进行误差修正，提高精度。
    
    参数:
        iterable: 可迭代对象，包含浮点数
    
    返回:
        高精度求和结果
    """
    sum_total = 0.0
    compensation = 0.0
    
    for num in iterable:
        # 减去之前的补偿值
        y = num - compensation
        # 累加，可能产生舍入误差
        t = sum_total + y
        # 计算新的补偿值
        compensation = (t - sum_total) - y
        # 更新总和
        sum_total = t
    
    return sum_total

def sum_neumaier(iterable):
    """
    Neumaier求和算法 - 比Kahan更精确
    
    算法原理:
        在Kahan算法基础上改进，当新项的绝对值大于当前和时，
        使用不同的计算顺序来减少误差。
    
    参数:
        iterable: 可迭代对象，包含浮点数
    
    返回:
        高精度求和结果
    """
    sum_total = 0.0
    compensation = 0.0
    
    for num in iterable:
        t = sum_total + num
        if abs(sum_total) >= abs(num):
            compensation += (sum_total - t) + num
        else:
            compensation += (num - t) + sum_total
        sum_total = t
    
    return sum_total + compensation

def sum_pairwise(iterable):
    """
    Pairwise求和算法 - 递归分治求和
    
    算法原理:
        将序列分成两半，分别求和后再相加，
        适用于大规模数据，精度最高。
    
    参数:
        iterable: 可迭代对象，包含数字
    
    返回:
        高精度求和结果
    """
    arr = list(iterable)
    
    if len(arr) == 0:
        return 0.0
    if len(arr) == 1:
        return float(arr[0])
    
    mid = len(arr) // 2
    left_sum = sum_pairwise(arr[:mid])
    right_sum = sum_pairwise(arr[mid:])
    
    return left_sum + right_sum

def sum_bigint(iterable):
    """
    大整数求和 - 支持任意精度
    
    参数:
        iterable: 可迭代对象，包含整数或字符串表示的大整数
    
    返回:
        Python大整数对象
    """
    result = 0
    for num in iterable:
        result += int(num)
    return result

# -*- coding: utf-8 -*-
# 并行求和模块

import multiprocessing as mp
import numpy as np
from .kahan import sum_kahan, sum_neumaier

def _sum_chunk(args):
    """
    内部函数：计算一个数据块的和
    
    参数:
        args: (data_chunk, algorithm) 元组
    
    返回:
        数据块的求和结果
    """
    data_chunk, algorithm = args
    
    if algorithm == 'kahan':
        return sum_kahan(data_chunk)
    elif algorithm == 'neumaier':
        return sum_neumaier(data_chunk)
    else:
        return sum(data_chunk)

def sum_parallel(iterable, workers=None, algorithm='basic'):
    """
    并行求和函数
    
    参数:
        iterable: 可迭代对象，包含数字
        workers: 进程数，默认为CPU核心数
        algorithm: 求和算法，可选 'basic', 'kahan', 'neumaier'
    
    返回:
        并行求和结果
    """
    # 确定进程数
    if workers is None:
        workers = mp.cpu_count()
    
    # 转换为列表
    data = list(iterable)
    n = len(data)
    
    # 如果数据量太小，直接求和
    if n <= workers * 1000:
        return sum(data)
    
    # 分割数据
    chunk_size = (n + workers - 1) // workers
    chunks = [data[i*chunk_size : (i+1)*chunk_size] 
              for i in range(workers)]
    
    # 创建进程池
    with mp.Pool(processes=workers) as pool:
        # 准备参数
        args = [(chunk, algorithm) for chunk in chunks]
        # 并行计算
        results = pool.map(_sum_chunk, args)
    
    # 合并结果
    if algorithm == 'kahan':
        return sum_kahan(results)
    elif algorithm == 'neumaier':
        return sum_neumaier(results)
    else:
        return sum(results)

def sum_numpy_parallel(array, workers=None):
    """
    使用NumPy进行并行求和
    
    参数:
        array: NumPy数组
        workers: 进程数，默认为CPU核心数
    
    返回:
        求和结果
    """
    if workers is None:
        workers = mp.cpu_count()
    
    arr = np.asarray(array)
    n = arr.size
    
    if n <= workers * 1000:
        return np.sum(arr)
    
    # 使用多线程(NumPy内部优化)
    np.seterr(all='ignore')
    result = np.sum(arr)
    
    return result
